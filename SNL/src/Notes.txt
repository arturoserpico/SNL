		succ(n) = n + 1;
		succ.inverse()(n) = n - 1;

		snl::Set::elements(zero, snl::all(n) | succ(n))
	}
}

	static Constructor<> zero;
	static Constructor<Integer> succ;

	Integer(int i) {
		if (i == 0)
			*this = zero;
		else
			*this = succ(Integer(i - 1));
	}
